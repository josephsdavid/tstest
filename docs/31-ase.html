tswase <- function(vec, ..., ahead) {
    f <- fore.arma.wge(vec, ..., n.ahead = ahead)
    ASE <- mean((f$f - vec[(length(vec) - (ahead - 1)):(length(vec))])^2)
    res <- list(fore = f, ASE = ASE)
    return(res)
}
